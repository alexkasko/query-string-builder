001    package com.alexkasko.springjdbc.querybuilder;
002    
003    /**
004     * Interface for expression lists. List is joined from expressions with commas.
005     * List is printed to output using {@link #toString()} method.
006     *
007     * @author alexkasko
008     * Date: 11/8/12
009     */
010    public interface ExpressionList {
011        /**
012         * Adds expression to list
013         *
014         * @param expr expression
015         * @return list itself
016         */
017        ExpressionList comma(Expression expr);
018    
019        /**
020         * Adds expression to list
021         *
022         * @param expr expression literal
023         * @return list itself
024         */
025        ExpressionList comma(String expr);
026    }
