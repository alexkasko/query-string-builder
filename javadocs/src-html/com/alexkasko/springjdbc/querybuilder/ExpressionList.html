001    package com.alexkasko.springjdbc.querybuilder;
002    
003    import java.util.Collection;
004    
005    /**
006     * Interface for expression lists. List is joined from expressions with commas.
007     * List is printed to output using {@link #toString()} method.
008     *
009     * @author alexkasko
010     * Date: 11/8/12
011     */
012    public interface ExpressionList {
013        /**
014         * Adds expression to list
015         *
016         * @param expr expression
017         * @return list itself
018         */
019        ExpressionList add(Expression expr);
020    
021        /**
022         * Adds expression to list
023         *
024         * @param expr expression literal
025         * @return list itself
026         */
027        ExpressionList add(String expr);
028    
029        /**
030         * Adds expressions collection to list
031         *
032         * @param exprs expressions collection
033         * @return list itself
034         */
035        ExpressionList add(Collection<Expression> exprs);
036    }
