001    package com.alexkasko.springjdbc.querybuilder;
002    
003    /**
004     * Static helper methods to work with expressions and expression lists
005     *
006     * @author alexkasko
007     * Date: 11/7/12
008     */
009    public final class Expressions {
010    
011        private Expressions() { }
012    
013        /**
014         * Creates expression list from string literal
015         *
016         * @param expr expression literal
017         * @return expression list
018         * @throws QueryBuilderException on empty input
019         */
020        public static ExpressionList list(String expr) {
021            return new ExprList(new LiteralExpr(expr));
022        }
023    
024        /**
025         * Creates expression from string literal
026         *
027         * @param expr expression literal
028         * @return expression
029         * @throws QueryBuilderException on empty input
030         */
031        public static Expression expr(String expr) {
032            return new LiteralExpr(expr);
033        }
034    
035        /**
036         * Creates negation expression for given expression
037         *
038         * @param expr expression
039         * @return negation expression
040         * @throws QueryBuilderException on null input
041         */
042        public static Expression not(Expression expr) {
043            return new NotExpr(expr);
044        }
045    
046        /**
047         * Creates negation expression for given expression literal
048         *
049         * @param expr expression literal
050         * @return negation expression
051         * @throws QueryBuilderException on empty input
052         */
053        public static Expression not(String expr) {
054            return new NotExpr(new LiteralExpr(expr));
055        }
056    
057        /**
058         * Creates disjunction expression for expressions
059         *
060         * @param exprs expressions for disjunction
061         * @return disjunction expression
062         * @throws QueryBuilderException on null input
063         */
064        public static Expression or(Expression... exprs) {
065            return new OrExpr(exprs);
066        }
067    }
