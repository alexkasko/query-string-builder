001    package com.alexkasko.springjdbc.querybuilder;
002    
003    /**
004     * Static helper methods to work with expressions and expression lists
005     *
006     * @author alexkasko
007     * Date: 11/7/12
008     */
009    public final class Expressions {
010    
011        private Expressions() { }
012    
013        /**
014         * Creates expression list
015         *
016         * @return expression list
017         */
018        public static ExpressionList list() {
019            return new ExprList();
020        }
021    
022        /**
023         * Creates expression list with prefix literal,
024         * that will be printed only if condition list is not empty
025         *
026         * @param prefix list prefix literal
027         * @return expression list
028         */
029        public static ExpressionList listWithPrefix(String prefix) {
030            return new ExprList(prefix);
031        }
032    
033        /**
034         * Creates expression list with {@code "order by"} prefix,
035         * that will be printed only if condition list is not empty
036         *
037         * @return expression list
038         */
039        public static ExpressionList orderBy() {
040            return new ExprList("order by");
041        }
042    
043        /**
044         * Creates expression from string literal
045         *
046         * @param expr expression literal
047         * @return expression
048         * @throws QueryBuilderException on empty input
049         */
050        public static Expression expr(String expr) {
051            return new LiteralExpr(expr);
052        }
053    
054        /**
055         * Creates negation expression for given expression
056         *
057         * @param expr expression
058         * @return negation expression
059         * @throws QueryBuilderException on null input
060         */
061        public static Expression not(Expression expr) {
062            return new NotExpr(expr);
063        }
064    
065        /**
066         * Creates negation expression for given expression literal
067         *
068         * @param expr expression literal
069         * @return negation expression
070         * @throws QueryBuilderException on empty input
071         */
072        public static Expression not(String expr) {
073            return new NotExpr(new LiteralExpr(expr));
074        }
075    
076        /**
077         * Creates disjunction expression for expressions
078         *
079         * @param exprs expressions for disjunction
080         * @return disjunction expression
081         * @throws QueryBuilderException on null input
082         */
083        public static Expression or(Expression... exprs) {
084            return new OrExpr(exprs);
085        }
086    
087        /**
088         * Creates prefix expression, prefix will be printed
089         * only if this expression will be conjuncted with other expressions,
090         * empty string will be printed otherwise
091         *
092         * @param prefix prefix literal
093         * @return prefix expression
094         * @throws QueryBuilderException on empty input
095         */
096        public static Expression prefix(String prefix) {
097            return  new PrefixExpr(prefix);
098        }
099    
100        /**
101         * Creates prefix expression with {@code "where"} prefix, that
102         * will be printed only if this expression will be conjuncted with other expressions,
103         * empty string will be printed otherwise
104         *
105         * @return prefix expression with {@code "where"} prefix
106         */
107        public static Expression where() {
108            return new PrefixExpr("where");
109        }
110    
111        /**
112         * Creates prefix expression with {@code "and"} prefix, that
113         * will be printed only if this expression will be conjuncted with other expressions,
114         * empty string will be printed otherwise
115         *
116         * @return prefix expression with {@code "and"} prefix
117         */
118        public static Expression and() {
119            return new PrefixExpr("and");
120        }
121    }
