001    package com.alexkasko.springjdbc.querybuilder;
002    
003    import org.apache.commons.lang.text.StrSubstitutor;
004    
005    import java.util.HashMap;
006    import java.util.Map;
007    import java.util.regex.Matcher;
008    import java.util.regex.Pattern;
009    
010    import static org.apache.commons.lang.StringUtils.isBlank;
011    
012    /**
013     * Builder class for string queries. Query is build from template - string with placeholder sections (clauses),
014     * and expressions (lists) set to to builder to fill clauses.
015     * Placeholder syntax: {@code ${placeholder_42}
016     * Placeholder escaping syntax: {@code $${placeholder_42} (will NOT be parsed)
017     * Placeholder name must conform this regex: {@code [a-zA-Z_0-9]+}
018     *
019     * @author alexkasko
020     * Date: 11/7/12
021     */
022    public class QueryBuilder {
023        private static final Pattern CLAUSE_PATTERN = Pattern.compile("[^\\$]\\$\\{(\\w+)\\}");
024    
025        private final String template;
026        private final Map<String, ExpressionList> clauses = new HashMap<String, ExpressionList>();
027    
028        /**
029         * Constructor
030         *
031         * @param template query template
032         */
033        public QueryBuilder(String template) {
034            if(isBlank(template)) throw new IllegalArgumentException("Provided template is blank");
035            this.template = template;
036            Matcher m = CLAUSE_PATTERN.matcher(template);
037            while (m.find()) {
038                String name = m.group(1);
039                if(clauses.containsKey(name)) throw new IllegalArgumentException(
040                        "Duplicate clause: '" + name +"' found in template: '" + template +"'");
041                clauses.put(name, null);
042            }
043        }
044    
045        /**
046         * Static method, invokes constructor
047         *
048         * @param template query template
049         * @return query vuilder instance
050         */
051        public static QueryBuilder query(String template) {
052            return new QueryBuilder(template);
053        }
054    
055        /**
056         * Registers provided string for clause
057         *
058         * @param clauseName clause name
059         * @param value clause value
060         * @return builder itself
061         */
062        public QueryBuilder set(String clauseName, String value) {
063            return set(clauseName, new LiteralExpr(value));
064        }
065    
066        /**
067         * Registers provided expression list for clause
068         *
069         * @param clauseName clause name
070         * @param expr clause expression
071         * @return builder itself
072         */
073        public QueryBuilder set(String clauseName, Expression expr) {
074            return set(clauseName, new ExprList(expr));
075        }
076    
077        /**
078         * Registers provided expression for clause
079         *
080         * @param clauseName clause name
081         * @param expr clause expression list
082         * @return builder itself
083         * @throws IllegalArgumentException on blank input, on missed clause, on duplicate close
084         */
085        public QueryBuilder set(String clauseName, ExpressionList expr) {
086            if(isBlank(clauseName)) throw new IllegalArgumentException("Provided clauseName is blank");
087            if(!clauses.containsKey(clauseName)) throw new IllegalArgumentException(
088                    "Provided clauseName: '" + clauseName + "' is not found in the template: '" +  template + "'" +
089                    " registered clauses: '" + clauses.keySet() + "'" +
090                    " (clause name must conform this regex: '[a-zA-Z_0-9]+')");
091            if(null != clauses.get(clauseName)) throw new IllegalArgumentException(
092                    "Provided clauseName: '" + clauseName + "' was already set to: '" + clauses.get(clauseName) + "'");
093            if(null == expr) throw new IllegalArgumentException("Provided expr is null");
094            clauses.put(clauseName, expr);
095            return this;
096        }
097    
098        /**
099         * Builds query string from template and accumulated values
100         *
101         * @return query string
102         * @throws IllegalStateException on not filled clause
103         */
104        public String build() {
105            Map<String, String> map = new HashMap<String, String>();
106            for(Map.Entry<String, ExpressionList> en : clauses.entrySet()) {
107                if(null == en.getValue()) throw new IllegalStateException(
108                        "Clause: '" + en.getKey() + "' wasn't filled, template: '" + template + "'");
109                map.put(en.getKey(), en.getValue().toString());
110            }
111            return StrSubstitutor.replace(template, map);
112        }
113    }
