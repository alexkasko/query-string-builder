001    package com.alexkasko.springjdbc.querybuilder;
002    
003    /**
004     * Interface for expressions.
005     * Expression is printed to output using {@link #toString()} method.
006     * All built-in expressions are immutable.
007     *
008     * @author alexkasko
009     * Date: 11/7/12
010     */
011    public interface Expression {
012        /**
013         * Adds another expression to this expression as a conjunction
014         *
015         * @param expr expression
016         * @return conjunction
017         */
018        Expression and(Expression expr);
019    
020        /**
021         * Adds another expression to this expression as a conjunction
022         *
023         * @param expr expression literal
024         * @return conjunction
025         */
026        Expression and(String expr);
027    }
